import cv2

# Load the pre-trained Haar Cascade Classifier for face detection
face_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_frontalface_default.xml')

# Function to detect faces in an image
def detect_faces("C:\Users\DELL\OneDrive\Pictures\Camera Roll\WIN_20240626_08_01_51_Pro.jpg"):
    # Read the image
    image = cv2.imread("C:\Users\DELL\OneDrive\Pictures\Camera Roll\WIN_20240626_08_01_51_Pro.jpg")
    # Convert to grayscale
    gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    # Detect faces
    faces = face_cascade.detectMultiScale(gray, scaleFactor=1.1, minNeighbors=5, minSize=(30, 30))
    
    # Draw rectangles around the faces
    for (x, y, w, h) in faces:
        cv2.rectangle(image, (x, y), (x+w, y+h), (255, 0, 0), 2)
    
    # Display the output
    cv2.imshow('Faces found',image)
    cv2.waitKey(0)
    cv2.destroyAllWindows()

# Example usage
detect_faces(r"C:\Users\DELL\OneDrive\Pictures\Camera Roll\WIN_20240626_08_01_51_Pro.jpg")
